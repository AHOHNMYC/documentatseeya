31static size_t
32blz_leparse_workmem_size(size_t src_size)
33{
34 return (LOOKUP_SIZE < 2 * src_size ? 3 * src_size : src_size + LOOKUP_SIZE)
35 * sizeof(blz_word);
36}

38// Backwards dynamic programming parse with left-extension of matches.
39//
40// Whenever we find a match that improves the cost at the current position,
41// we try to extend this match to the left, and if possible we use that
42// left-extension for each position to the left. Since we are processing
43// the input from right to left, this matches repeated patterns without
44// searching at each position.
45//
46// Essentially, this improves the worst case for the parsing at a small cost
47// in ratio. The match finding is still O(n^2) in number of matches though,
48// so may have to limit max_depth on larger block sizes.
49//
50// This is usually within a few percent of the "optimal" parse with the same
51// parameters.
52//
53static unsigned long
54blz_pack_leparse(const void *src, void *dst, unsigned long src_size, void *workmem,
55 const unsigned long max_depth, const unsigned long accept_len)
56{
57 struct blz_state bs;
58 const unsigned char *const in = (const unsigned char *) src;
59 const unsigned long last_match_pos = src_size > 4 ? src_size - 4 : 0;
60
61 assert(src_size < BLZ_WORD_MAX);
62
63 // Check for empty input
64 if (src_size == 0) {
65 return 0;
66 }
67
68 bs.next_out = (unsigned char *) dst;
69
70 // First byte verbatim
71 *bs.next_out++ = in[0];
72
73 // Check for 1 byte input
74 if (src_size == 1) {
75 return 1;
76 }
77
78 // Initialize first tag
79 bs.tag_out = bs.next_out;
80 bs.next_out += 2;
81 bs.tag = 0;
82 bs.bits_left = 16;
83
84 if (src_size < 4) {
85 for (unsigned long i = 1; i < src_size; ++i) {
86 // Output literal tag
87 blz_putbit(&bs, 0);
88
89 // Copy literal
90 *bs.next_out++ = in[i];
91 }
92
93 // Return compressed size
94 return (unsigned long) (blz_finalize(&bs) - (unsigned char *) dst);
95 }
96
97 // With a bit of careful ordering we can fit in 3 * src_size words.
98 //
99 // The idea is that the lookup is only used in the first phase to
100 // build the hash chains, so we overlap it with mpos and mlen.
101 // Also, since we are using prev from right to left in phase two,
102 // and that is the order we fill in cost, we can overlap these.
103 //
104 // One detail is that we actually use src_size + 1 elements of cost,
105 // but we put mpos after it, where we do not need the first element.
106 //
107 blz_word *const prev = (blz_word *) workmem;
108 blz_word *const mpos = prev + src_size;
109 blz_word *const mlen = mpos + src_size;
110 blz_word *const cost = prev;
111 blz_word *const lookup = mpos;
112
113 // Phase 1: Build hash chains
114 const int bits = 2 * src_size < LOOKUP_SIZE ? BLZ_HASH_BITS : blz_log2(src_size);
115
116 // Initialize lookup
117 for (unsigned long i = 0; i < (1UL << bits); ++i) {
118 lookup[i] = NO_MATCH_POS;
119 }
120
121 // Build hash chains in prev
122 if (last_match_pos > 0) {
123 for (unsigned long i = 0; i <= last_match_pos; ++i) {
124 const unsigned long hash = blz_hash4_bits(&in[i], bits);
125 prev[i] = lookup[hash];
126 lookup[hash] = i;
127 }
128 }
129
130 // Initialize last three positions as literals
131 mlen[src_size - 3] = 1;
132 mlen[src_size - 2] = 1;
133 mlen[src_size - 1] = 1;
134
135 cost[src_size - 3] = 27;
136 cost[src_size - 2] = 18;
137 cost[src_size - 1] = 9;
138 cost[src_size] = 0;
139
140 // Phase 2: Find lowest cost path from each position to end
141 for (unsigned long cur = last_match_pos; cur > 0; --cur) {
142 // Since we updated prev to the end in the first phase, we
143 // do not need to hash, but can simply look up the previous
144 // position directly.
145 unsigned long pos = prev[cur];
146
147 assert(pos == NO_MATCH_POS || pos < cur);
148
149 // Start with a literal
150 cost[cur] = cost[cur + 1] + 9;
151 mlen[cur] = 1;
152
153 unsigned long max_len = 3;
154
155 const unsigned long len_limit = src_size - cur;
156 unsigned long num_chain = max_depth;
157
158 // Go through the chain of prev matches
159 for (; pos != NO_MATCH_POS && num_chain--; pos = prev[pos]) {
160 unsigned long len = 0;
161
162 // If next byte matches, so this has a chance to be a longer match
163 if (max_len < len_limit && in[pos + max_len] == in[cur + max_len]) {
164 // Find match len
165 while (len < len_limit && in[pos + len] == in[cur + len]) {
166 ++len;
167 }
168 }
169
170 // Extend current match if possible
171 //
172 // Note that we are checking matches in order from the
173 // closest and back. This means for a match further
174 // away, the encoding of all lengths up to the current
175 // max length will always be longer or equal, so we need
176 // only consider the extension.
177 if (len > max_len) {
178 unsigned long min_cost = ULONG_MAX;
179 unsigned long min_cost_len = 3;
180
181 // Find lowest cost match length
182 for (unsigned long i = max_len + 1; i <= len; ++i) {
183 unsigned long match_cost = blz_match_cost(cur - pos - 1, i);
184 assert(match_cost < BLZ_WORD_MAX - cost[cur + i]);
185 unsigned long cost_here = match_cost + cost[cur + i];
186
187 if (cost_here < min_cost) {
188 min_cost = cost_here;
189 min_cost_len = i;
190 }
191 }
192
193 max_len = len;
194
195 // Update cost if cheaper
196 if (min_cost < cost[cur]) {
197 cost[cur] = min_cost;
198 mpos[cur] = pos;
199 mlen[cur] = min_cost_len;
200
201 // Left-extend current match if possible
202 if (pos > 0 && in[pos - 1] == in[cur - 1]) {
203 do {
204 --cur;
205 --pos;
206 ++min_cost_len;
207 unsigned long match_cost = blz_match_cost(cur - pos - 1, min_cost_len);
208 assert(match_cost < BLZ_WORD_MAX - cost[cur + min_cost_len]);
209 unsigned long cost_here = match_cost + cost[cur + min_cost_len];
210 cost[cur] = cost_here;
211 mpos[cur] = pos;
212 mlen[cur] = min_cost_len;
213 } while (pos > 0 && in[pos - 1] == in[cur - 1]);
214 break;
215 }
216 }
217 }
218
219 if (len >= accept_len || len == len_limit) {
220 break;
221 }
222 }
223 }
224
225 mpos[0] = 0;
226 mlen[0] = 1;
227
228 // Phase 3: Output compressed data, following lowest cost path
229 for (unsigned long i = 1; i < src_size; i += mlen[i]) {
230 if (mlen[i] == 1) {
231 // Output literal tag
232 blz_putbit(&bs, 0);
233
234 // Copy literal
235 *bs.next_out++ = in[i];
236 }
237 else {
238 const unsigned long offs = i - mpos[i] - 1;
239
240 // Output match tag
241 blz_putbit(&bs, 1);
242
243 // Output match length
244 blz_putgamma(&bs, mlen[i] - 2);
245
246 // Output match offset
247 blz_putgamma(&bs, (offs >> 8) + 2);
248 *bs.next_out++ = offs & 0x00FF;
249 }
250 }
251
252 // Return compressed size
253 return (unsigned long) (blz_finalize(&bs) - (unsigned char *) dst);
254}
